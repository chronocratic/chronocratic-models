    **Step 2 - Create decorators.py (GREEN):**

    ```python
    V = TypeVar("V", covariant=True)  # Note: Seeded needs its own TypeVar for Generic

    class Seeded(Generic[V]):
        """Deterministic decorator wrapping a stateless AugmentationProducer.

        Constraint: Must NOT wrap TrainableAugmentationProducer.

        Uses torch.random.fork_rng() to isolate seed from outer context.
        """
        def __init__(self, *, inner: AugmentationProducer[V], seed: int) -> None:
            if isinstance(inner, TrainableAugmentationProducer):
                raise TypeError(
                    "Seeded cannot wrap TrainableAugmentationProducer. "
                    "Stateless producers only (SPEC §4.6)."
                )
            self._inner, self._seed = inner, seed

        def produce(self, x: torch.Tensor) -> V:
            with torch.random.fork_rng():
                torch.manual_seed(self._seed)
                return self._inner.produce(x)
    ```

    **Step 3 - Create trainable_support.py (GREEN):**

    Import `AugmentationProducer`, `TrainableAugmentationProducer`, `SingleView` from `base.py`. Import `nn` from `torch`, `Any` from `typing`.

    ```python
    def maybe_train_augmentation(
        augmentation: AugmentationProducer[Any],
        *, x: torch.Tensor, encoder: nn.Module, epoch: int, batch_idx: int,
    ) -> torch.Tensor | None:
        if not isinstance(augmentation, TrainableAugmentationProducer):
            return None
        if not augmentation.should_train_augmentation(epoch=epoch, batch_idx=batch_idx):
            return None
        return augmentation.train_step(x=x, encoder=encoder, batch_idx=batch_idx)

    def maybe_configure_augmentation_optimizer(
        augmentation: AugmentationProducer[Any], *, lr: float,
    ) -> torch.optim.Optimizer | None:
        if not isinstance(augmentation, TrainableAugmentationProducer):
            return None
        return augmentation.configure_optimizer(lr=lr)
    ```
